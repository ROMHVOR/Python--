from typing import List, Optional, Tuple, Dict
import heapq
from collections import deque
from ..graph.graph import Graph


class ShortestPath:
    """Класс для поиска кратчайших путей"""
    
    @staticmethod
    def dijkstra(graph: Graph, start_id: str, end_id: Optional[str] = None) -> Tuple[Dict[str, float], Dict[str, Optional[str]]]:
        """
        Алгоритм Дейкстры для поиска кратчайших путей во взвешенном графе
        
        Args:
            graph: Граф (должен быть взвешенным)
            start_id: ID начальной вершины
            end_id: ID конечной вершины (опционально)
            
        Returns:
            Tuple[Dict[str, float], Dict[str, Optional[str]]]: 
                - Словарь расстояний до каждой вершины
                - Словарь предшественников для восстановления пути
        """
        if start_id not in graph._nodes:
            raise ValueError(f"Vertex {start_id} not found in graph")
        
        # Проверяем, есть ли веса у ребер
        has_weights = any(graph._weights.values())
        if not has_weights:
            raise ValueError("Graph has no edge weights. Use BFS for unweighted graphs.")
        
        distances = {node_id: float('inf') for node_id in graph._nodes}
        predecessors = {node_id: None for node_id in graph._nodes}
        distances[start_id] = 0
        
        priority_queue = [(0, start_id)]
        visited = set()
        
        while priority_queue:
            current_dist, current_id = heapq.heappop(priority_queue)
            
            if current_id in visited:
                continue
            
            visited.add(current_id)
            
            if end_id and current_id == end_id:
                break
            
            for neighbor in graph.get_neighbors(current_id):
                if neighbor.id in visited:
                    continue
                
                weight = graph.get_edge_weight(current_id, neighbor.id)
                if weight is None:
                    continue
                
                distance = current_dist + weight
                
                if distance < distances[neighbor.id]:
                    distances[neighbor.id] = distance
                    predecessors[neighbor.id] = current_id
                    heapq.heappush(priority_queue, (distance, neighbor.id))
        
        return distances, predecessors
    
    @staticmethod
    def bfs_shortest_path(graph: Graph, start_id: str, end_id: str) -> Optional[List[str]]:
        """
        Поиск кратчайшего пути в невзвешенном графе с помощью BFS
        
        Args:
            graph: Невзвешенный граф
            start_id: ID начальной вершины
            end_id: ID конечной вершины
            
        Returns:
            Optional[List[str]]: Список вершин пути или None, если путь не найден
        """
        if start_id not in graph._nodes or end_id not in graph._nodes:
            raise ValueError("Start or end vertex not found in graph")
        
        if start_id == end_id:
            return [start_id]
        
        visited = {start_id}
        queue = deque([(start_id, [start_id])])
        
        while queue:
            current_id, path = queue.popleft()
            
            for neighbor in graph.get_neighbors(current_id):
                if neighbor.id == end_id:
                    return path + [neighbor.id]
                
                if neighbor.id not in visited:
                    visited.add(neighbor.id)
                    queue.append((neighbor.id, path + [neighbor.id]))
        
        return None  # Путь не найден
    
    @staticmethod
    def get_path(predecessors: Dict[str, Optional[str]], start_id: str, end_id: str) -> List[str]:
        """
        Восстановить путь из словаря предшественников
        
        Args:
            predecessors: Словарь предшественников
            start_id: ID начальной вершины
            end_id: ID конечной вершины
            
        Returns:
            List[str]: Список вершин пути
        """
        path = []
        current = end_id
        
        while current is not None:
            path.append(current)
            current = predecessors.get(current)
            if current == start_id:
                path.append(current)
                break
        
        path.reverse()
        return path if path and path[0] == start_id else []
