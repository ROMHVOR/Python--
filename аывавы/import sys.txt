    def handle_choice(self, choice: str):
        """Обработать выбор пользователя"""
        actions = {
            '1': self.create_graph,
            '2': self.add_node,
            '3': self.add_edge,
            '4': self.remove_node,
            '5': self.remove_edge,
            '6': self.show_graph,
            '7': self.traverse_bfs,
            '8': self.traverse_dfs,
            '9': self.find_shortest_path,
            '10': self.save_graph,
            '11': self.load_graph,
            '12': self.show_info,
            '0': self.exit_app,
        }
        
        action = actions.get(choice)
        if action:
            try:
                action()
            except Exception as e:
                print(f"\n❌ Ошибка: {e}")
        else:
            print("\n❌ Неверный выбор. Пожалуйста, выберите от 0 до 12.")
